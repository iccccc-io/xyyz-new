  const axios = require('axios');

  });

  const _ = db.command;

  const QWEN_API_URL = 'https://dashscope.aliyuncs.com/compatible-mode/v1/chat/completions';
  
  exports.main = async (event, context) => {
    const { post_id } = event;
    if (!post_id) return { success: false, msg: '缺少 post_id 参数' };
  
    try {
      // 1. 获取帖子详情
      const postRes = await db.collection('community_posts').doc(post_id).get();
      const post = postRes.data;
  
      // 检查管线状态，防止重复执行
      if (post.pipeline_status !== 'pending') {
        return { success: false, msg: '该帖子管线状态不为 pending' };
      }
  
      // 更新状态为处理中
      await db.collection('community_posts').doc(post_id).update({
        data: { pipeline_status: 'processing' }
      });
  
      const images = post.images || [];
      const content = post.content || '无配文';
      
      // 提取关联的非遗项目名称 (处理不同数据结构兼容性)
      let related_names = '';
      if (post.related_projects && post.related_projects.length > 0) {
        related_names = post.related_projects.map(p => p.name).join('、');
      } else {
        related_names = '未关联特定项目';
      }
  
      // 2. 提取需要处理的 cloud:// 图片 ID 列表
      const fileList = [];
      images.forEach((img, index) => {
        if (img.process_status === 0 && img.url.startsWith('cloud://')) {
          fileList.push(img.url);
        }
      });
  
      // 如果没有需要处理的图片，直接跳过视觉解析阶段
      if (fileList.length === 0) {
        return { success: true, msg: '无待处理图片，管线继续' };
      }
  
      // 3. 将 cloud:// 转换为 https:// 临时链接
      const tempUrlRes = await cloud.getTempFileURL({ fileList: fileList });
      const tempUrlMap = {}; // { 'cloud://...': 'https://...' }
      tempUrlRes.fileList.forEach(item => {
        if (item.status === 0) { // 0 表示获取成功
          tempUrlMap[item.fileID] = item.tempFileURL;
        }
      });
  
      // 4. 串行调用大模型解析图片 (控制并发，防止被阿里封控)
      for (let i = 0; i < images.length; i++) {
        const img = images[i];
        
        // 只处理待处理的图片
        if (img.process_status !== 0) continue;
  
        const tempUrl = tempUrlMap[img.url];
        if (!tempUrl) {
          // 获取临时链接失败，标记状态为 -1
          await updateImageStatus(post_id, i, -1, '获取临时链接失败');
          continue;
        }

        const systemPrompt = `你是一个专业的非遗社区运营专家。你收到的是用户发布的帖子中的配图，结合用户配文：【${content}】以及该帖子关联的非遗项目：【${related_names}】，分析这张图片。
  任务：用一句话（50字以内）精准描述图片信息。
  限制：直接输出描述文本，绝不允许输出任何解释性废话（如'好的'、'这张图是'等开头）。`;
  
        const requestBody = {
          model: "qwen-vl-plus",
          messages: [
            {
              role: "user",
              content: [
                { type: "image_url", image_url: { url: tempUrl } },
                { type: "text", text: systemPrompt }
              ]
            }
          ]
        };
  
        try {
          // 发起 HTTP 请求调用通义千问
          const aiRes = await axios.post(QWEN_API_URL, requestBody, {
            headers: {
              'Authorization': `Bearer ${QWEN_API_KEY}`,
              'Content-Type': 'application/json'
            },
            timeout: 15000 // 设置 15 秒超时
          });
  
          // 提取 AI 描述
          const ai_desc = aiRes.data.choices[0].message.content.trim();
  
          // 5. 数据库精准回写 (使用点操作符)
          await updateImageStatus(post_id, i, 1, ai_desc);
          console.log(`[成功] 图片 ${i} 解析完成`);
  
        } catch (aiError) {

          await updateImageStatus(post_id, i, -1, '大模型解析失败');
        }
      }
  
      // 6. （预留）下一步：同步 Dify 知识库逻辑
      // 当循环结束后，所有图片解析完毕，可以执行同步 Dify 的逻辑
      // ... 调用 dify 同步逻辑 ...
        // 6. 触发下一步：同步 Dify 知识库逻辑
      // 视觉解析循环结束后，无论图片解析是成功(1)还是失败(-1)，只要管线没崩溃，就必须进入 Dify 同步环节。
      try {
        await cloud.callFunction({
          name: 'sync_dify_knowledge',
          data: { post_id: post_id }
        });

        // 仅标记同步失败，不影响前面已成功的图片解析结果

          data: { pipeline_status: 'sync_failed' }
        });
      }
  
      return { success: true, msg: '视觉解析结束，已移交 Dify 同步管线' };
  
    } catch (error) {
      console.error('管线全局报错:', error);
      // 将管线状态标记为异常
      await db.collection('community_posts').doc(post_id).update({
        data: { pipeline_status: 'failed' }
      });
      return { success: false, error: error.message };
    }
  };
  
  // 辅助函数：原子化更新单张图片状态
  async function updateImageStatus(postId, index, status, desc) {
    const db = cloud.database();
    const updateData = {
      [`images.${index}.process_status`]: status
    };
    
    // 只有成功时才更新描述
    if (status === 1) {
      updateData[`images.${index}.ai_description`] = desc;
    }
  
    await db.collection('community_posts').doc(postId).update({

    });
  }

  ###### 代码逻辑说明 (致开发组)：

  1.  **临时链接映射 (`tempUrlMap`)**：云函数使用 `cloud.getTempFileURL` 批量获取外链。因为原数组 `images` 可能混杂了非 `cloud://` 的脏数据，所以采用了 Map 字典来精准对应 `fileID` 和 `HTTPS` 链接，确保大模型能读取。
  2.  **严格的串行机制 (`for...of`)**：这是保护你 API Key 的关键。代码使用了 `for` 循环搭配 `await`，保证前一张图解析完（或失败）后，才开始请求第二张图，绝不并发。
  3.  **原子更新函数 (`updateImageStatus`)**：这是防覆盖的核心。它动态拼接类似 `'images.0.process_status'` 的键名，保证每次更新只修改指定的字段，不会破坏数组中的其他图片数据。
  4.  **容错降级**：哪怕某一张图解析报错（大模型超时），代码会捕获异常将该图标记为 `-1`，然后**继续执行下一张图**，绝不让整个管线崩溃死锁。

##### 2.3 数据库精准回写

* **防覆盖更新**：
  拿到单张图片的解析结果后，**绝对禁止**将整个 `images` 数组查出来修改后再整体覆盖覆盖（会导致并发下的数据丢失）。
  **必须使用点操作符定位更新：**

  ```javascript
  db.collection('community_posts').doc(post_id).update({
    data: {
      [`images.${index}.ai_description`]: result_text,
      [`images.${index}.process_status`]: 1
    }
  })
  ```

  let dify_text = `【标题】：${post.title}\n【正文】：${post.content}\n`;

#### 附：社区内容实时同步 Dify 规范

###### 1. 架构目标与策略

传统的 RAG 知识库是静态的，而我们的社区（UGC 帖子）是动态生长的。为了让 AI 随时掌握最新的社区经验，系统将采取**“文档级映射（Document Mapping）”**的同步策略：

*   **每一篇**审核通过的社区帖子，在 Dify 知识库 `UGC_Community_Posts` 中对应一个独立的 `Document`。
*   通过小程序的**云函数**监听数据库的变动，实时触发 Dify 的增/删/改 API。

---

###### 2. 环境配置与安全前提

*   **后端节点**：`https://api.dify.ai/v1`
*   **目标知识库**：`UGC_Community_Posts`
*   **API Key 配置**：
    *   由于是“知识库 API”，你需要为该特定知识库生成专用的 API Key。
    *   **开发要求**：严禁将 Key 写死在代码中，必须配置在云函数的**环境变量 (`DIFY_KNOWLEDGE_API_KEY`)** 中。
    *   **Dataset ID**：通过调用一次 `GET /datasets` 接口获取该知识库的 `dataset_id`，并配置在环境变量中。

---

###### 3. 数据库扩展改造 (`community_posts` 集合)

为了能够精准地在 Dify 中找到对应的帖子进行更新或删除，我们设计在小程序的 `community_posts` 集合中新增一个“外键”：

*   **`dify_document_id`** (String)：
    *   初始为 `null`。
    *   当云函数第一次将帖子推送到 Dify 成功后，将 Dify 返回的 `document.id` 存入此字段。

---

###### 4. 核心同步流水线设计 (Sync Pipelines)

开发组需编写一个名为 `sync_dify_knowledge` 的核心云函数，处理以下三种场景：

##### 场景 A：发布新帖 (Create)

* **触发条件**：用户发布帖子，且（若有视觉解析）视觉解析已完成。

* **Dify API**：`POST /datasets/{dataset_id}/document/create_by_text`

* **请求体 (Body) 构建规则**：

  ```json
  {
    "name": "post_p12345.txt", // 建议用 post_id 命名，方便排查
    "text": "标题：[帖子标题]\n正文：[帖子正文]\n工艺描述：[AI视觉解析描述]\n[包含的非遗项目]...", // 必须拼装完整的上下文
    "indexing_technique": "high_quality", // 必须指定！
    "process_rule": { "mode": "automatic" }
  }
  ```

* **回调处理 (Callback)**：

  *   接口调用成功后，会返回 `"id": "dify返回的文档ID"`。
  *   **必须**将该 ID 回写到小程序数据库该帖子的 `dify_document_id` 字段。

  另：**HTTP Header 要求（必带）**：

  ```JSON
  {
    "Authorization": "Bearer {配置在环境变量中的 DIFY_API_KEY}",
    "Content-Type": "application/json"
  }
  ```

##### 场景 B：删除/私密化帖子 (Delete)

*   **触发条件**：帖子被用户删除、被管理员下架、或被设为“仅自己可见”。
*   **前置检查**：检查该帖子是否有 `dify_document_id`，若有则执行下一步。
*   **Dify API**：`DELETE /datasets/{dataset_id}/documents/{document_id}`
    *   *注：此处的 `{document_id}` 就是我们存的 `dify_document_id`。*
*   **预期结果**：返回 `204 No Content`。该内容瞬间从 AI 脑海中抹除，防止隐私泄露。

##### 场景 C：修改帖子 (Update)

* **触发条件**：用户修改了帖子内容。

* **前置检查**：提取 `dify_document_id`。若为空，走**场景 A (Create)**。

* **Dify API**：`POST /datasets/{dataset_id}/documents/{document_id}/update_by_text`

* **请求体 (Body)**：

  ```json
  {
    "name": "post_p12345.txt",
    "text": "[修改后的最新完整文本]" 
  }
  ```

* **逻辑说明**：Dify 会自动丢弃旧文本的向量，重新对新文本进行 Embedding，保证 AI 搜到的是最新版本。

  **场景 C 补充说明：包含新图片的编辑逻辑**

  - 
  - **前端要求**：如果用户在编辑帖子时上传了**新图片**，前端在 update 数据库时，必须将新图片的 process_status 初始化为 0，并将帖子的 pipeline_status 重新设为 "pending"。
  - **后端路由**：
    - 
    - 若 pipeline_status == 'pending'（说明有新图片）：后台必须重新触发 trigger_ai_pipeline。视觉管线会自动跳过旧图片（状态为 1），只解析新图片（状态为 0），解析完后再调 Dify 的 update_by_text 接口。
    - 若只是纯文字修改：直接触发 sync_dify_knowledge 云函数，调用 Dify 的 update_by_text 接口覆盖旧文本即可。

###### 5. 避坑指南与错误处理 (Error Handling)

开发组在对接 Dify API 时，请重点关注以下异常处理机制：

1.  **异步与超时设计**：
    Dify 的文本处理和向量化需要时间。调用 `create` 或 `update` 后，返回的 `"indexing_status"` 为 `"waiting"` 是正常现象。
    **不要让云函数轮询等待！** 只要接口 HTTP 状态码为 200/202，即可结束云函数，让 Dify 后台自行处理。
2.  **失败重试补偿**：
    若因为网络抖动导致 Dify API 报 `500` 或超时，必须有**重试机制**（或标记为 `sync_failed` 稍后由定时器补传），坚决杜绝“小程序里有，但 AI 搜不到”的数据黑洞。

## 请确保有以下 异常容错与降级策略

在分布式异步系统中，随时可能断网或接口超时。开发组必须实现以下兜底机制：

1. **大模型超时降级**：
   * 阿里云接口调用需设置 `timeout: 15000` (15秒)。若超时或抛错，将该图片的 `process_status` 设为 `-1`。
2. **Dify 同步失败处理**：
   * 若 Dify API 返回 502/504，必须在云函数内实现 `try-catch` 捕获，并将帖子的 `pipeline_status` 标记为 `sync_failed`。
3. **超大图片处理**：
   * 若用户上传了 10MB 的原图，直接发给大模型会报错且浪费极多 Token。要求开发组在使用 `cloud.getTempFileURL` 之前，通过云调用的图片处理能力，将图片动态压缩（如加上后缀缩放参数）后再传给 AI。
